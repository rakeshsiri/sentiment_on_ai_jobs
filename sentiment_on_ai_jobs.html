#After dropping null values, 0, and excluding 20th percent values on both sides. Mean is calculated country wise.
sns.set(rc={'figure.figsize':(10,5)})
def mean_(x):
    '''
    Calculates mean for given array, after excluding nulls and outliers (20% on both ends)
    INPUT: numpy array
    
    OPUTPUT: int
    
    '''
    x = x.dropna()
    x = x[x!=0]
    x = x[(x.quantile(.2)<x) & (x < x.quantile(.8))]
    return x.mean()

country_data.groupby('Country')['ConvertedSalary'].apply(lambda x:mean_(x)).sort_values(ascending=True).plot('barh',color=sns.color_palette()[0],title = 'Average Salary by country');

def get_ai_sentiment_on(col):
    '''
    Multiple Bar Graph for plotting general sentiment on AI for 4 different categories
    
    INPUT: string - one of the four following columns
    'AIFuture', 'AIResponsible','AIDangerous','AIInteresting'
    
    OUTPUT: None
    '''
    country_data[col] = country_data[col].fillna('No comments')
    opinion_order = country_data[col].dropna().unique()
    col_name = 'scale_to_100_'+col
    age_order = sorted(list(country_data['YearsCodingProf'].drop_duplicates().dropna().values),key=lambda x:int(x.split('-')[0] if '-' in x else 30));
    data = country_data.groupby('YearsCodingProf')[col].apply(lambda x:x.value_counts()/len(x)*100).reset_index(name=col_name).rename(columns={'level_1':'opinions'})
    g = sns.barplot(hue = 'opinions', x = 'YearsCodingProf', y = col_name, data = data,
                order = age_order, hue_order = opinion_order);#color = sns.color_palette()[0]
    g.set_xticklabels(rotation=30,labels = age_order);
    g.set_title(col)
